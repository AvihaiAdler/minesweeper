add_library(tigr STATIC)

find_library(GLU
  NAMES GLU
  REQUIRED
)

find_library(X11
  NAMES X11
  REQUIRED
)

find_library(GL
  NAMES GL
  REQUIRED
)

target_sources(tigr PRIVATE
  tigr.c
)

target_compile_options(tigr PRIVATE
  -Wall
  -Wextra
  -pedantic
)

target_link_libraries(tigr PUBLIC
  ${GLU}
  ${X11}
  ${GL}
)
